a = int(input("Начало диапазона: "))
b = int(input("Конец диапазона: "))
evens = []  
odds = []  
nines = [] 
for c in range(a, b + 1):
    if c % 2 == 0:
        evens.append(c)
    else:
        odds.append(c)
    if c % 9 == 0:
        nines.append(c)
def avg(lst):
    return sum(lst) / len(lst) if lst else 0
print("Четные:", evens)
print("Среднее:", avg(evens))
print("Нечетные:", odds)
print("Среднее:", avg(odds))
print("Кратные 9:", nines)
print("Среднее:", avg(nines))
